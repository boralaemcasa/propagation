	function dobro(x: my_type4) return my_type4 is
	variable y: my_type4;
	variable z: my_type12;
	begin
	  y(0) := not(x(0));
	  y(1) := not(x(1));

	  y(3) := not(x(3));

	  y(1) := z(1);
	  y(2) := z(2);
	  y(3) := z(3);
	  return y;
	end dobro;

begin
  Q <= std_logic_vector(dobro(my_type4(D)));
